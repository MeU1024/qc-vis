```
  ├── components
  │  │   eventBus.ts

  │  │   viewer.ts

  │         qcviewermanager.ts
  │         qcviewerpanel.ts

  │      VisPanel.ts	
  ├── providers
  │  │   abstraction.ts

  │  │   structure.ts

  │  │      abstractionrule.ts     
  │  └── structurelib

  │         layout.ts			
  │         qcmodel.ts             
  │         quantumgate.ts 
  ├── utilities

```
